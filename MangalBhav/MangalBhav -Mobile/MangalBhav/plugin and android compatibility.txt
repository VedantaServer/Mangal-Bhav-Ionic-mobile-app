Installing the plugin (thumb machine)

1) Create plugin 
   -> npx cap generate plugin fingerprint-plugin

2) Create .tgz file for the plugin
   -> cd ./pluginfolder
   -> npm run build
   -> npm pack

3) Installing the .tgz file in the main directory.
   -> cd..
   -> npm install .\fingerprint-bridge\vedantacollege-fingerprint-bridge-0.0.1.tgz

android {
    namespace "mobile.facupai.com"
    compileSdk rootProject.ext.compileSdkVersion
    defaultConfig {
        applicationId "mobile.facupai.com"
        minSdkVersion rootProject.ext.minSdkVersion
        targetSdkVersion rootProject.ext.targetSdkVersion
        versionCode 1
        versionName "1.0"
        testInstrumentationRunner "androidx.test.runner.AndroidJUnitRunner"
        aaptOptions {
            // Files and dirs to omit from the packaged assets dir...
            ignoreAssetsPattern '!.svn:!.git:!.ds_store:!*.scc:.*:!CVS:!thumbs.db:!picasa.ini:!*~'
        }
    }

    // <- ADD THIS BLOCK (exactly)
    packagingOptions {
        jniLibs {
            excludes += ["**/libbarhopper_v3.so", "**/libimage_processing_util_jni.so"]
        }
    }
    // <- end added block

    buildTypes {
        release {
            minifyEnabled false
            proguardFiles getDefaultProguardFile('proguard-android.txt'), 'proguard-rules.pro'
        }
    }
}
